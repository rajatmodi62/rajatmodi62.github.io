<script>
document.addEventListener("DOMContentLoaded", function() {
    const system = document.getElementById('bond-video-system');
    const overlay = document.getElementById('boot-overlay');
    const vid = document.getElementById('lab-core-video');

    if (system && vid) {
        system.addEventListener('click', function() {
            vid.style.display = 'block';
            
            // Ensure video is UNMUTED before playing
            vid.muted = false; 
            
            const playPromise = vid.play();

            if (playPromise !== undefined) {
                playPromise.then(() => {
                    overlay.style.display = 'none';
                }).catch(error => {
                    console.warn("Audio play blocked by browser. Muting to allow visual playback.");
                    // Fallback: if browser STILL blocks it, we mute so it at least plays
                    vid.muted = true;
                    vid.play();
                    overlay.style.display = 'none';
                });
            }
        }, { once: true });
    }
});
</script>
